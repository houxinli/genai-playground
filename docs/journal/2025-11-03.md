## 操作记录（主要命令）
```bash
# 本地 Provider（Ollama）
export LLM_PROVIDER=ollama
export LLM_BASE_URL=http://localhost:11434
export LLM_API_KEY=ollama

# 健康检查（可指向远程或本地 OpenAI 兼容端口）
python scripts/check_vllm.py http://localhost:11434

# 更新/创建 mac 环境
conda env update -n llm -f environment-llm-mac.yml
conda run -n llm python -c "import pixivpy3, bs4, html5lib, openai; print('ok')"

# Pixiv 批量下载（示例作者 50235390）
python tasks/translation/src/scripts/batch_download_v1.py \
  --user-id 50235390 \
  --output-root tasks/translation/data \
  --limit 50 --offset 0 --workers 2 --rate-limit 1 --retries 5

# 翻译（Ollama + Qwen3 4B 指令量化）
./tasks/translation/translate tasks/translation/data/pixiv/50235390 \
  --bilingual-simple --stream \
  --llm-provider ollama --llm-base-url http://localhost:11434 \
  --model qwen3:4b-instruct-2507-q4_K_M
```
